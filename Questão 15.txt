#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
    char texto[100];
    int i,c, contar=0;
    printf("Digete umas palavras ou letras aleatorias para saber quantos |a| tem\n");
    scanf(" %s", texto);
    c = strlen(texto);
    for(i=0;texto[i]!= '\0'; i++ ) {
        if(texto[i]=='a') {
        contar++;
        }
    }printf("\nnumeros de a = %d\n", contar);

    return 0;
}
